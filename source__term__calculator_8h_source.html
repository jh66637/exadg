34template<int dim, typename Number>
35class SourceTermCalculator
36{
37 using This = SourceTermCalculator<dim, Number>;
38 using VectorType = dealii::LinearAlgebra::distributed::Vector<Number>;
39
40 using CellIntegratorScalar = CellIntegrator<dim, 1, Number>;
41 using CellIntegratorVector = CellIntegrator<dim, dim, Number>;
42
43public:

46 }
47
48 void
49 setup(dealii::MatrixFree<dim, Number> const & matrix_free_in,
50 SourceTermCalculatorData const & data_in)
51 {
52 matrix_free = &matrix_free_in;
53 data = data_in;
54 }
55
56 void
57 evaluate_integrate(VectorType & dst,
58 VectorType const & velocity_cfd_in,
59 VectorType const & pressure_cfd_in,
60 VectorType const & pressure_cfd_time_derivative_in)
61 {
62 dst.zero_out_ghost_values();
63
64 if(data.consider_convection)
65 {
66 velocity_cfd.reset(velocity_cfd_in);
67 velocity_cfd->update_ghost_values();
68
69 pressure_cfd.reset(pressure_cfd_in);
70 pressure_cfd->update_ghost_values();
71 }
72
73 matrix_free->cell_loop(
74 &This::compute_source_term, this, dst, pressure_cfd_time_derivative_in, true);
75 }

77private:
78 void
79 compute_source_term(dealii::MatrixFree<dim, Number> const & matrix_free_in,
80 VectorType & dst,
81 VectorType const & dp_cfd_dt,
82 std::pair<unsigned int, unsigned int> const & cell_range) const
83 {
84 CellIntegratorScalar dpdt(matrix_free_in, data.dof_index_pressure, data.quad_index);
85 CellIntegratorScalar p(matrix_free_in, data.dof_index_pressure, data.quad_index);
86 CellIntegratorVector u(matrix_free_in, data.dof_index_velocity, data.quad_index);
87
88 Number rho = static_cast<Number>(data.density);
89
90 for(unsigned int cell = cell_range.first; cell < cell_range.second; ++cell)
91 {
92 dpdt.reinit(cell);
93 dpdt.gather_evaluate(dp_cfd_dt, dealii::EvaluationFlags::values);
94
95 if(data.consider_convection)
96 {
97 p.reinit(cell);
98 p.gather_evaluate(*pressure_cfd, dealii::EvaluationFlags::gradients);
99 u.reinit(cell);
100 u.gather_evaluate(*velocity_cfd, dealii::EvaluationFlags::values);
101
102 for(unsigned int q = 0; q < dpdt.n_q_points; ++q)
103 dpdt.submit_value(-rho * (dpdt.get_value(q) + u.get_value(q) * p.get_gradient(q)), q);
104 }
105 else
106 {
107 for(unsigned int q = 0; q < dpdt.n_q_points; ++q)
108 dpdt.submit_value(-rho * dpdt.get_value(q), q);
109 }
110
111 dpdt.integrate_scatter(dealii::EvaluationFlags::values, dst);
112 }
113 }
114

121};
